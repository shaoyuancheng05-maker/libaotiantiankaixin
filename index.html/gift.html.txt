<script src="https://cdn.jsdelivr.net/npm/canvas-confetti@1.6.0/dist/confetti.browser.min.js"></script>
<script>
  // 掉落的负面情绪词汇，你也可以自己增加
  const words = ["压力", "烦躁", "委屈", "疲惫", "焦虑", "生气", "emo"];
  const container = document.getElementById('game-container');
  const scoreEl = document.getElementById('score');
  const winScreen = document.getElementById('win-screen');
  let score = 0;
  const targetScore = 15; // 点击15次即可通关
  let spawnInterval;

  function spawnBadMood() {
    const el = document.createElement('div');
    el.className = 'bad-mood';
    el.innerText = words[Math.floor(Math.random() * words.length)];
    // 随机水平位置，避免超出屏幕边缘
    el.style.left = Math.random() * (window.innerWidth - 70) + 'px';
    el.style.top = '-70px';
    
    // 下落速度在 3 到 5 秒之间随机
    const duration = Math.random() * 2 + 3; 
    el.style.animationDuration = duration + 's';

    // 点击事件
    el.ontouchstart = el.onmousedown = function(e) {
      e.preventDefault(); // 防止手机端双击放大
      burst(el.offsetLeft, el.offsetTop); // 触发爱心特效
      el.remove(); // 移除方块
      score++;
      scoreEl.innerText = score;
      if(score >= targetScore) { win(); }
    };

    container.appendChild(el);
    // 掉出屏幕后自动清理DOM
    setTimeout(() => { if (el.parentElement) el.remove(); }, duration * 1000);
  }

  function burst(x, y) {
    const emojis = ['🌸', '❤️', '✨', '😘'];
    const burstEl = document.createElement('div');
    burstEl.className = 'emoji-burst';
    burstEl.innerText = emojis[Math.floor(Math.random() * emojis.length)];
    burstEl.style.left = (x + 20) + 'px';
    burstEl.style.top = y + 'px';
    container.appendChild(burstEl);
    setTimeout(() => burstEl.remove(), 800);
  }

  function win() {
    clearInterval(spawnInterval); // 停止掉落
    container.innerHTML = ''; // 清空画面剩余方块
    winScreen.style.display = 'flex'; // 显示表白页面
    
    // 触发两侧持续撒花特效
    var duration = 4 * 1000;
    var end = Date.now() + duration;
    (function frame() {
      confetti({ particleCount: 5, angle: 60, spread: 55, origin: { x: 0 }, colors: ['#ffb6c1', '#ff69b4', '#ff1493', '#fff'] });
      confetti({ particleCount: 5, angle: 120, spread: 55, origin: { x: 1 }, colors: ['#ffb6c1', '#ff69b4', '#ff1493', '#fff'] });
      if (Date.now() < end) { requestAnimationFrame(frame); }
    }());
  }

  // 每 0.7 秒掉落一个
  spawnInterval = setInterval(spawnBadMood, 700);
</script>
